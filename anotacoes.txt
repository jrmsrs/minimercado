Criar uma logica do banco de dados similar a:

  /* ---- Limpa/cria tabelas ---- */

    /* Catálogo */
    DROP TABLE IF EXISTS produto;

    /* Venda */
    DROP TABLE IF EXISTS venda;

    /* Catálogo */
    CREATE TABLE IF NOT EXISTS produto(
      prod_id int AUTO_INCREMENT PRIMARY KEY, 
      nome text NOT NULL,
      preco real NOT NULL
    );

    /* Venda */
    CREATE TABLE IF NOT EXISTS venda(
      venda_id int AUTO_INCREMENT PRIMARY KEY, 
      fk_prod_id int,
      qtd int NOT NULL,
      data text NOT NULL,
      FOREIGN KEY (fk_prod_id) REFERENCES produto(prod_id)
    );

  /* ---- Insere nas tabelas ---- */
    /* Catálogo */
    insert into `produto` (nome, preco, prod_id) values ('toddy', 6, 5);
    insert into `produto` (nome, preco, prod_id) values ('pilao', 8, 4);
    insert into `produto` (nome, preco, prod_id) values ('oreo', 3, 3);
    insert into `produto` (nome, preco, prod_id) values ('coca', 4, 2);
    insert into `produto` (nome, preco, prod_id) values ('nescau', 7, 1);

    /* Venda */
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/1/22', 1, 3, 1);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/1/22', 3, 1, 2);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/2/22', 2, 1, 3);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/2/22', 1, 2, 4);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/2/22', 5, 2, 5);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/2/22', 3, 1, 6);
    insert into `venda` (data, fk_prod_id, qtd, venda_id) values ('1/3/22', 4, 1, 7);

  /* ---- Consulta as tabelas ---- */
    /* Catálogo */
    SELECT * FROM produto;

    /* Venda */
    SELECT * FROM venda;

    /* Relatório */
    SELECT v.data, sum(p.preco*v.qtd) total
    FROM venda v
    LEFT JOIN produto p 
    ON v.fk_prod_id=p.prod_id
    GROUP BY v.data;
